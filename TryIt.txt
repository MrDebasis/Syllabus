handleSave(event) {
    const objectName = event.target.dataset.obj;
    const draftValues = event.detail.draftValues;

    console.log(`🔄 Save triggered for object: ${objectName}`);
    console.log('✏️ Draft values:', JSON.stringify(draftValues, null, 2));

    // Update local data with draft values
    this.tables[objectName].data = this.tables[objectName].data.map(row => {
        const draft = draftValues.find(d => d.Id === row.Id);
        return draft ? { ...row, ...draft } : row;
    });

    // Clear draft values after save
    this.tables[objectName].draftValues = [];

    // Log updated data for verification
    console.log(`✅ Updated data for ${objectName}:`, JSON.stringify(this.tables[objectName].data, null, 2));
}
