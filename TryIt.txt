@track _imageIds = new Set();

@api
get imageIds() {
    return [...this._imageIds]; // Convert Set to Array for template reactivity
}

set imageIds(value) {
    if (Array.isArray(value)) {
        this._imageIds = new Set(value); // Ensure uniqueness when setting an array
    } else if (value) {
        this._imageIds.add(value); // Add a single unique value
    }
}
