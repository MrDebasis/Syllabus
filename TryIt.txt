getHighlightFlags(row) {
    const flags = {};

    const nextActionDueDateVal = row['NextActionDueDate__c'];
    if (nextActionDueDateVal) {
        const dueDate = new Date(nextActionDueDateVal);
        const today = new Date(); today.setHours(0, 0, 0, 0);
        if (dueDate > today) {
            flags.isNextActionDueDateRed = true;
        }
    }

    return flags;
}
