
   @wire(getFeatureAnnouncement)
    wiredgetFeatureAnnouncement({ error, data }) {
        if (data) {
            if (data && typeof data === 'object') {
                Object.keys(data).forEach(key => {
                    this.userDetailedMap.set(key, data[key]);
            });
            this.setValues();
        }

        } else if (error) {
            console.error(error);
        }
    }
