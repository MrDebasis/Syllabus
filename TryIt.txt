<template if:true={isNotSureYetSelected}>
    <label>When should we check back in to confirm if bridging therapy will be needed?*</label>
    <lightning-input type="date"
        value={notSureYetDate}
        onchange={handleNotSureDateChange}>
    </lightning-input>
</template>



@track notSureYetDate = '';

handleNotSureDateChange(event) {
    this.notSureYetDate = event.target.value;

    // Store the date in answers
    const uniqueId = 'BridgingTherapyDate_NotSureYet';
    const label = 'Bridging Therapy Not Sure Yet - Follow Up Date';

    const newAnswer = {
        UniqueId: uniqueId,
        label: label,
        value: this.notSureYetDate
    };

    const existingIndex = this.answers.findIndex(a => a.UniqueId === uniqueId);
    if (existingIndex > -1) {
        this.answers.splice(existingIndex, 1, newAnswer);
    } else {
        this.answers.push(newAnswer);
    }

    this.isNextDisabled = !this.notSureYetDate;
    console.log('Not Sure Yet Date stored:', newAnswer);
}
