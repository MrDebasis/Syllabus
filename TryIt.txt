set contentDocumentIds(value) {
    if (Array.isArray(value)) {
        this._contentDocumentIds = value;
    } else {
        this._contentDocumentIds = [value];
    }

    this.initializeImageMap();
    
    // Filter allFiles to keep only those that have a matching ContentDocumentId
    this.allFiles = this.allFiles.filter(file => 
        this._contentDocumentIds.includes(file.ContentDocumentId)
    );

    console.debug('Filtered allFiles:', this.allFiles);
}
