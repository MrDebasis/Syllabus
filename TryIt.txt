customSortByNextActionFields(cases) {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const isEmpty = val => val === null || val === undefined || val === '' || String(val).trim() === '';

    const getRank = record => {
        const isNextActionEmpty = isEmpty(record.NextAction__c);
        const isDueDateEmpty = isEmpty(record.NextActionDueDate__c);
        const dueDate = record.NextActionDueDate__c ? new Date(record.NextActionDueDate__c) : null;
        const isFutureDate = dueDate && dueDate > today;

        if (isNextActionEmpty && isDueDateEmpty) {
            return 0; // Highest priority
        } else if (isFutureDate) {
            return 1; // Middle priority
        } else {
            return 2; // Lowest priority
        }
    };

    return cases.sort((a, b) => {
        return getRank(a) - getRank(b);
    });
}
