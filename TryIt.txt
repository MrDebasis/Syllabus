 public Integer compareTo(Object compareTo) {
    Notifications compareToRecord = (Notifications)compareTo;
    if (this.CreatedDate < compareToRecord.CreatedDate) {
        return 1;   // This record comes after the compared record
    } else if (this.CreatedDate > compareToRecord.CreatedDate) {
        return -1;  // This record comes before the compared record
    } else {
        return 0;   // They are equal
    }
}
