function formatDateIfValid(inputDate) {
    // Define a regular expression to match the 'Jun 28, 2022' format
    const datePattern = /^[A-Za-z]{3} \d{2}, \d{4}$/;

    // Check if the inputDate matches the pattern
    if (datePattern.test(inputDate)) {
        // If the date matches, create a Date object and format it as yyyy-mm-dd
        let dateObj = new Date(inputDate);

        // Get the correct UTC date components
        let year = dateObj.getUTCFullYear();
        let month = (dateObj.getUTCMonth() + 1).toString().padStart(2, '0'); // Months are zero-based
        let day = dateObj.getUTCDate().toString().padStart(2, '0');

        // Return the formatted date
        return `${year}-${month}-${day}`;
    } else {
        // If the date doesn't match, return the input as is
        return inputDate;
    }
}

// Example Usage
let date1 = "Jun 28, 2022";
let date2 = "2022-06-28";

console.log(formatDateIfValid(date1)); // Output: 2022-06-28
console.log(formatDateIfValid(date2)); // Output: 2022-06-28 (unchanged)
