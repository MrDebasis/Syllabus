get associatedContact() {
    const table = this.tables['HealthCloudGA__ContactContactRelation__c'];
    if (!table) return null;

    // Add accountUrl field to each record in .data
    const enhancedData = table.data.map(record => ({
        ...record,
        accountUrl: record.Account__c
            ? `/lightning/r/Account/${record.Account__c}/view`
            : null
    }));

    return {
        ...table,
        data: enhancedData
    };
}
