import { LightningElement, track } from 'lwc';

export default class UniqueRecordsComponent extends LightningElement {
    _unsubmittedArray = []; // Private variable to store the array

    get unsubmittedArray() {
        return this._unsubmittedArray;
    }

    set unsubmittedArray(value) {
        // Ensure unique records based on 'id' when setting the array
        this._unsubmittedArray = [...new Map(value.map(item => [item.id, item])).values()];
    }

    // Example method to add a new item
    addItem(newItem) {
        this.unsubmittedArray = [...this._unsubmittedArray, newItem]; // Triggers setter
    }
}
