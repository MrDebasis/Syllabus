// Convert all values in filter.value to Date objects
const dateArray = filter.value.map(v => new Date(v));

// Find the earliest and latest dates
const start = new Date(Math.min(...dateArray));
const end = new Date(Math.max(...dateArray));

// Compare valDate with the start and end range
return valDate >= start && valDate <= end;
