                    this.getConvertedFormat(contentDocumentId)
                        .then((convertedDocumentId) => {
                            if (convertedDocumentId) {
                                this.strContentDocId.push(convertedDocumentId);
                                this.finalizeUpload();

                            }
                        })

                        });
