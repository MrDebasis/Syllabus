      console.log('connectedCallback started');
        this.subscribeToMessageChannel();
        console.log('Subscribed to message channel');
        this.publishMessage();
        console.log('Message published');
    }

    publishMessage() {
        const message = { messageId: 'test 1234' };
        console.log('Preparing to publish message:', JSON.stringify(message));
        publish(this.messageContext, SAMPLE_MESSAGE_CHANNEL, message);
        console.log('Message published successfully');
    }

    subscribeToMessageChannel() {
        try {
            console.log('Subscribing to message channel...');
            subscribe(this.messageContext, SAMPLE_MESSAGE_CHANNEL, (message) => {
                console.log('Message received:', JSON.stringify(message));
            });
        } catch (error) {
            console.error('Error while subscribing:', error);
        }
    }
