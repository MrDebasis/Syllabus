    @api isTutorial =false;

    @api hideArrow = false;
    @api pageName= 'ReimbursementPage';
    @api currentScreenIndex;
    @track flowScreens = {
        'ReimbursementPage' : 
            {
               'STEP 2' : {bool : 'reimbursementPageToturialStep_2'}, 
               'STEP 3' : {bool : 'reimbursementPageToturialStep_3' }
            }
    }

    handleBtn(){
        if(this.pageName === 'ReimbursementPage' && this.btnLabel ==='Next'){
            this.closeToolTipPopUp =true;
             const sendToParent =this.flowScreens[this.pageName][this.currentScreenIndex].bool ;
            this.dispatchEvent(new CustomEvent('handlenextbtn', {
                detail: {
                    message: sendToParent
                }
            }));
        }
        if(this.btnLabel==='Finish'){
            this.handleClose();
      }
    }

    handleClose(){
        this.closeToolTipPopUp =false;
        this.dispatchEvent(new CustomEvent('handlenextbtn', {
