convertDateToYYYYMMDD(inputDate) {
    // Parse the input date (e.g., "September 3, 2024")
    const parsedDate = new Date(inputDate);

    // Check if the date is valid
    if (isNaN(parsedDate)) {
        console.error('Invalid date format');
        return null;
    }

    // Get year, month, and day
    const year = parsedDate.getFullYear();
    const month = String(parsedDate.getMonth() + 1).padStart(2, '0'); // Months are zero-indexed, so add 1
    const day = String(parsedDate.getDate()).padStart(2, '0'); // Pad single-digit days with a leading zero

    // Return formatted date in 'YYYY-MM-DD'
    return `${year}-${month}-${day}`;
}
