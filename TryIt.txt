if (record.hasOwnProperty("AdditionalInfoNeeded__c")) {
    const additionalInfo = record.AdditionalInfoNeeded__c.split(";"); // Split the additional info into an array
    const notesList = [];

    additionalInfo.forEach((info) => {
        const matchingConfig = this.expenseStatusConfig.find(
            (config) => config.AdditionalInformationNeeded__c === info
        );
        if (matchingConfig) {
            notesList.push(matchingConfig.PortalNotes__c);
        }
    });

    // Join the notes into a string and assign them to the Notes__c field
    record.Notes__c = notesList.join(", ");
}
