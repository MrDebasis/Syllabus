function calculateTimeAgo(createdDate, todaysDate) {
    const created = new Date(createdDate);
    const now = new Date(todaysDate);

    // Check for valid dates
    if (isNaN(created.getTime()) || isNaN(now.getTime())) {
        return 'Invalid date';
    }

    const diffInSeconds = Math.floor((now - created) / 1000); // Difference in seconds

    if (diffInSeconds < 60) {
        return 'Now'; // Less than a minute ago
    }

    const diffInMinutes = Math.floor(diffInSeconds / 60);
    if (diffInMinutes < 60) {
        return `${diffInMinutes} minute${diffInMinutes > 1 ? 's' : ''} ago`;
    }

    const diffInHours = Math.floor(diffInMinutes / 60);
    if (diffInHours < 24) {
        return `${diffInHours} hour${diffInHours > 1 ? 's' : ''} ago`;
    }

    const diffInDays = Math.floor(diffInHours / 24);
    if (diffInDays < 30) {
        return `${diffInDays} day${diffInDays > 1 ? 's' : ''} ago`;
    }

    const diffInMonths = Math.floor(diffInDays / 30);
    if (diffInMonths < 12) {
        return `${diffInMonths} month${diffInMonths > 1 ? 's' : ''} ago`;
    }

    const diffInYears = Math.floor(diffInMonths / 12);
    return `${diffInYears} year${diffInYears > 1 ? 's' : ''} ago`;
}
