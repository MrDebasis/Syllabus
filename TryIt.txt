
    convertDateFormat(dateString) {
        const date = new Date(dateString);
        return new Intl.DateTimeFormat('en-US', { month: 'short', day: 'numeric', year: 'numeric' }).format(date);
      }
	  
	  
    async prePopulateValuesWhenReceptIsUpdated(versionId) {
        if(this.files[0].Id !==null && this.files[0].Id !=='' && this.files[0].Id !== undefined){
        try {
            const result = await getReceiptJSON({ contentVersionId: versionId });
            let values = JSON.parse(result);
            this.vendor =  this.vendor !==null &&  this.vendor !=='' &&  this.vendor !== undefined ?  this.vendor  : values.vendorName;
            this.expenseDate =  this.expenseDate !==null &&  this.expenseDate !=='' &&  this.expenseDate !== undefined ?  this.expenseDate  : this.convertDateFormat(values.transactionDate);
            this.cityOfPurchage =  this.cityOfPurchage !==null &&  this.cityOfPurchage !=='' &&  this.cityOfPurchage !== undefined ?  this.cityOfPurchage  : values.cityOfPurchase;
            this.expenseAmount =  this.expenseAmount !==null &&  this.expenseAmount !=='' &&  this.expenseAmount !== undefined ?  this.expenseAmount  : values.total;
            this.validateSaveButton();
        } catch (error) {
            console.error(error);
        }
     }
