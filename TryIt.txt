 updatedRow = this.updateNextActionCellHighlight(updatedRow);

updateNextActionCellHighlight(row) {
    const nextActionCol = row.columns.find(col => col.field === 'nextaction__c');
    const dueDateCol = row.columns.find(col => col.field === 'nextactionduedate__c');

    if (!nextActionCol || !dueDateCol) return row;

    const dueDate = dueDateCol.value ? new Date(dueDateCol.value) : null;
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const isFutureDue = dueDate && dueDate > today;
    const isNextActionBlank = [null, '', undefined].includes(nextActionCol.value);

    nextActionCol.cellClass = (isFutureDue && isNextActionBlank) ? 'highlight-yellow' : '';

    return row;
}
