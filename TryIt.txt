handleRefresh() {
    console.log('Refresh called', filterLogic.filters);

    refreshApex(this.wiredCasesResult)
        .then(() => {
            // Now this.cases is updated after refresh
            if (filterLogic.filters.length > 0) {
                this.casesFiltered = this.cases.filter(caseItem =>
                    filterLogic.filters.every(filter =>
                        this.evaluateField(caseItem, filter)
                    )
                );
            } else {
                this.casesFiltered = [...this.cases]; // no filters, return all
            }
        })
        .catch(error => {
            console.error('Error refreshing data:', error);
        });
}
